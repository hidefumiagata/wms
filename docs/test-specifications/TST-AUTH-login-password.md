### SC-009: アカウントロック（5回失敗→ロック）

```typescript
import { test, expect } from '@playwright/test';

test('SC-009: 5回連続ログイン失敗でアカウントがロックされる', async ({ page }) => {
  await page.goto('/login');

  // Step 1-4: 4回連続で誤ったパスワードでログイン
  for (let i = 0; i < 4; i++) {
    await page.fill('[data-testid="user-code"]', 'lock_target');
    await page.fill('[data-testid="password"]', 'Wrong@1234');
    await page.click('[data-testid="login-submit"]');

    // 認証失敗メッセージが表示される
    await expect(page.locator('[data-testid="error-banner"]')).toContainText(
      'ユーザーコードまたはパスワードが正しくありません'
    );

    // フォームをクリアして次の試行へ
    await page.fill('[data-testid="user-code"]', '');
    await page.fill('[data-testid="password"]', '');
  }

  // Step 5: 5回目の失敗でアカウントロック
  await page.fill('[data-testid="user-code"]', 'lock_target');
  await page.fill('[data-testid="password"]', 'Wrong@1234');
  await page.click('[data-testid="login-submit"]');

  // ロックメッセージが表示される
  await expect(page.locator('[data-testid="error-banner"]')).toContainText(
    'アカウントがロックされました。管理者に連絡してください'
  );

  // Step 6: 正しいパスワードでもログイン不可
  await page.fill('[data-testid="user-code"]', 'lock_target');
  await page.fill('[data-testid="password"]', 'Target@1234');
  await page.click('[data-testid="login-submit"]');

  await expect(page.locator('[data-testid="error-banner"]')).toContainText(
    'アカウントがロックされました'
  );

  // ログイン画面に留まることを確認
  await expect(page).toHaveURL(/\/login/);
});
```
